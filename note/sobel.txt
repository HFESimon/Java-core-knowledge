img_rgb = imread('pout.tif');
img = rgb2gray(img_rgb);
subplot(121);
imshow(img);
a_bmp = filter2(fspecial('average',3),img)/255; 
imwrite(a_bmp,'A.bmp');
bw1 = edge(a_bmp,'sobel');
p1 =fspecial('sobel');
subplot(122);
add1 =imfilter(a_bmp,p1);
out1=add1+a_bmp;
imshow(out1);
title('sobel锐化结果');
